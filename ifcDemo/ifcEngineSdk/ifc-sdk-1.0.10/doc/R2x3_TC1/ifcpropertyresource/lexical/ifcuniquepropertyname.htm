FUNCTION IfcUniquePropertyName
(Properties : SET [1:?] OF IfcProperty)
:LOGICAL;

LOCAL
  Names : SET OF IfcIdentifier := [];
END_LOCAL;

REPEAT i:=1 TO HIINDEX(Properties);
  Names := Names + Properties[i].Name;
END_REPEAT;
RETURN (SIZEOF(Names) = SIZEOF(Properties));
END_FUNCTION;
